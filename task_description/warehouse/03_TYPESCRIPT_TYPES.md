## Довідники

```typescript
// ── Склад ─────────────────────────────────────────────────
export interface Warehouse {
  id: number;
  name: string;
  isActive: boolean;
}

// ── Категорія / Підкатегорія ──────────────────────────────
export interface Category {
  id: number;
  name: string;
}

export interface Subcategory {
  id: number;
  categoryId: number;
  name: string;
  category?: Category;
}

// ── Тип витрат ────────────────────────────────────────────
export type CostTypeCode = "fixed" | "variable";

export interface CostType {
  id: number;
  name: string;           // «Фіксована» / «Змінна» — для відображення
  code: CostTypeCode;      // явний код замість порівняння name-рядка
}

// ── Об'єкт витрат ─────────────────────────────────────────
export interface ExpenseObject {
  id: number;
  warehouseId: number;
  categoryId: number;
  subcategoryId: number;
  name: string;
  costTypeId: number;
  isActive: boolean;
  // join-поля (з API)
  warehouse?: Warehouse;
  category?: Category;
  subcategory?: Subcategory;
  costType?: CostType;
}

export type ExpenseObjectCreate = Omit<
  ExpenseObject,
  "id" | "warehouse" | "category" | "subcategory" | "costType"
>;
```

## Записи витрат

```typescript
export interface ExpenseRecord {
  id: number;
  expenseObjectId: number;
  year: number;
  month: number;               // 1–12
  amount: number;
  comment?: string;
  isAutoGenerated: boolean;
  createdAt: string;
  updatedAt: string;
  // join
  expenseObject?: ExpenseObject;
}

export type ExpenseRecordCreate = Omit<
  ExpenseRecord,
  "id" | "isAutoGenerated" | "createdAt" | "updatedAt" | "expenseObject"
>;

// ── Історія фіксованих витрат ─────────────────────────────
export interface FixedExpenseHistoryEntry {
  id: number;
  expenseObjectId: number;
  amount: number;
  startDate: string;           // ISO date, з якої діє сума
  comment?: string;
  createdAt: string;
  createdByName: string;
}

export type FixedExpenseHistoryCreate = Omit<
  FixedExpenseHistoryEntry,
  "id" | "createdAt" | "createdByName"
>;

// ── Результат каскадного оновлення (§4.3 в overview) ──────
export interface CascadeUpdateResult {
  updatedRecord: ExpenseRecord;
  cascadedCount: number;        // скільки авто-згенерованих майбутніх записів оновлено
}

// ── Резолюція суми фіксованого об'єкта на місяць ──────────
export type FixedResolutionStatus = "current" | "future" | "fallback" | "none";

export interface FixedAmountResolution {
  expenseObjectId: number;
  status: FixedResolutionStatus;
  amount: number | null;        // null тільки при status === "none"
  effectiveDate: string | null; // start_date відповідного запису історії
  comment: string;               // напр. «Актуальна з 01.03.2026»
  isEditable: boolean;           // true лише коли status === "none"
}
```

// ── Рядок форми масового внесення за місяць ───────────────
export interface MonthlyEntryRow {

export interface YearGridCell {

  amount: number | null;

export interface YearGridData {
  warehouseId: number;
  year: number;
  expenseObjects: ExpenseObject[];
  cells: YearGridCell[];          // expenseObjects.length × 12
}

export interface YearGridSaveResult {
  updatedCount: number;
  deletedCount: number;          // клітинки, звільнені до 0/порожньо
}

export interface DashboardKpis {

export interface MonthlyTrendPoint {
  year: number;

  fixedUah: number;
  variableUah: number;
}

// ── Розбивка по категоріях ────────────────────────────────
export interface CategoryBreakdown {
  categoryId: number;
  categoryName: string;

  sharePct: number;
}

// ── Порівняння складів ────────────────────────────────────
export interface WarehouseComparison {
  warehouseId: number;
  warehouseName: string;
  totalUah: number;
  fixedUah: number;
  variableUah: number;
  objectsCount: number;
}

// ── Рік-до-року ────────────────────────────────────────────
export interface YoyComparisonPoint {
  month: number;                 // 1–12
  years: Record<number, number>; // { 2025: 41200, 2026: 45830 }
}

// ── План / факт ────────────────────────────────────────────
export interface BudgetVsActualRow {
  categoryId: number;
  categoryName: string;
  plannedUah: number;   // сума останньої fixed_expense_history (по fixed-об'єктах)
                          // + trailing 3-міс. середнє (по variable-об'єктах)
  actualUah: number;
  deltaUah: number;
  deltaPct: number;
}

// ── Прогноз наступного місяця ─────────────────────────────
export interface ForecastPoint {
  categoryId: number;
  categoryName: string;
  forecastUah: number;
  basis: "fixed_history" | "trailing_average";
}

// ── Аномалія ───────────────────────────────────────────────
export interface AnomalyFlag {
  warehouseId: number;
  year: number;
  month: number;
  totalUah: number;
  trailingAverageUah: number;
  deviationPct: number;          // (total - avg) / avg * 100
}

// ── Річна деталізація (аналог analysis.html) ──────────────
export interface YearlyAnalysisData {
  monthlyData: MonthlyTrendPoint[];
  yearlyTotals: { year: number; totalUah: number }[];
  categoryBreakdown: CategoryBreakdown[];
  totalYearlyUah: number;
  recordsCount: number;
  monthsCount: number;
  avgMonthlyUah: number;
}
